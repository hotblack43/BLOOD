Plot Weight and Blood-pressure

Setups

rm(list=ls())
library("readxl")
library(varhandle)

Notes

May have to first convert the xls files to csv with e.g.ssconvert pressure.xls pressure.csv

get Weight data

weight <- read.csv( "~/WORKSHOP/BP/DATA/weight.csv")
weight <- unfactor(weight)
weight$Weight <- as.numeric(gsub("[a-z\\()]","",weight$Weight))
# POSIX
myPOSIX <- as.data.frame(within(weight, { timestamp=strptime(paste(Date,Time), "%Y/%m/%d%H:%M:%S") }))
df <- cbind(myPOSIX$timestamp,weight)
df <- df[,c("myPOSIX$timestamp","Weight")]

get BP Pressure data

pressure <- read.csv( "~/WORKSHOP/BP/DATA/pressure.csv")
pressure <- unfactor(pressure)
SYS <- pressure$SYS.mmHg
DIA <- pressure$DIA.mmHg
PULSE <- pressure$Pulse.Beats.Min
# POSIX
myPOSIX <- as.data.frame(within(pressure, { timestamp=strptime(paste(Date,Time), "%Y/%m/%d%H:%M:%S") }))
df2 <- myPOSIX[,c(-1,-2)]

Plots

par(mfrow=c(3,2))
plot(df,xlab="Date",ylab="Weight [kg]")
abline(h=100,col=2,lwd=3)
plot(df2$timestamp,df2$SYS.mmHg,xlab="Date",ylab="Systolic [mm Hg]")
plot(df2$timestamp,df2$DIA.mmHg,xlab="Date",ylab="Diastolic [mm Hg]")
plot(df2$timestamp,df2$Pulse.Beats.Min,xlab="Date",ylab="Pulse [bpm]")
#
plot(df2$Pulse.Beats.Min,df2$SYS.mmHg,xlab="Pulse",ylab="Systolic")
